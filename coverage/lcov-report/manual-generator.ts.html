import * as fs from 'fs';
import * as path from 'path';
import { Logger } from './logger';
import { TemplateManager } from './template-manager';
import { TemplateContext } from './types';
 
export class ManualGenerator {
  private isProcessing = false;
  private logger: Logger;
  private templateManager: TemplateManager;
  private outputDir: string;
 
  constructor() {
    this.logger = Logger.getInstance();
    this.templateManager = new TemplateManager();
    this.outputDir = path.join(process.cwd(), 'output');
  }
 
  async generateManuals(
    context: TemplateContext
  ): Promise<{ mainOutput: string; controlOutput: string }> {
    if (this.isProcessing) {
      this.logger.warn('매뉴얼 생성이 이미 진행 중입니다');
      throw new Error('매뉴얼 생성이 이미 진행 중입니다');
    }
 
    this.isProcessing = true;
    this.logger.info('매뉴얼 생성 시작');
 
    try {
      // 출력 디렉토리 생성
      if (!fs.existsSync(this.outputDir)) {
        fs.mkdirSync(this.outputDir, { recursive: true });
        this.logger.debug('출력 디렉토리 생성됨', { outputDir: this.outputDir });
      }
 
      // 메인 매뉴얼 생성
      this.logger.debug('메인 매뉴얼 템플릿 로드 시작');
      const mainTemplate = await this.templateManager.loadTemplates('main');
      this.logger.debug('메인 매뉴얼 템플릿 로드 완료');
 
      this.logger.debug('메인 매뉴얼 컨텍스트 처리 시작');
      const mainContent = this.templateManager.processTemplateWithContext(mainTemplate, context);
      this.logger.debug('메인 매뉴얼 컨텍스트 처리 완료');
 
      const mainOutput = path.join(this.outputDir, 'main.txt');
      fs.writeFileSync(mainOutput, mainContent);
      this.logger.debug('메인 매뉴얼 파일 생성 완료', { mainOutput });
 
      // 제어 매뉴얼 생성
      this.logger.debug('제어 매뉴얼 템플릿 로드 시작');
      const controlTemplate = await this.templateManager.loadTemplates('control');
      this.logger.debug('제어 매뉴얼 템플릿 로드 완료');
 
      this.logger.debug('제어 매뉴얼 컨텍스트 처리 시작');
      const controlContent = this.templateManager.processTemplateWithContext(
        controlTemplate,
        context
      );
      this.logger.debug('제어 매뉴얼 컨텍스트 처리 완료');
 
      const controlOutput = path.join(this.outputDir, 'control.txt');
      fs.writeFileSync(controlOutput, controlContent);
      this.logger.debug('제어 매뉴얼 파일 생성 완료', { controlOutput });
 
      this.logger.info('매뉴얼 생성 완료', { mainOutput, controlOutput });
 
      return { mainOutput, controlOutput };
    } catch (error) {
      this.logger.error('매뉴얼 생성 실패', {
        error: error instanceof Error ? error.message : String(error),
      });
      throw error;
    } finally {
      this.isProcessing = false;
      this.logger.debug('매뉴얼 생성 프로세스 종료');
    }
  }
 
  public getLogs(): Array<{ level: string; message: string; details?: Record<string, unknown> }> {
    return this.logger.getLogs();
  }
 
  public clearLogs(): void {
    this.logger.clearLogs();
  }
}
